#include <stdio.h>
//15 20 0 5 -1
//10 20
int main(void)
{
    int sum = 0;
    int num;
    int avg;
    int max = 0;
    int count = 0;

    while (1)
    {   //taking user input
        scanf("%d", &num);
        if (num < 0)
            break;
        sum = sum + num;
        count++;
        if (num > max)
        {
            max = num;
        }
    }
    //calculating average
    avg = sum / count;
    //printing average and max
    printf("%d %d\n", avg, max);
    return 0;
}

#include <stdio.h>
int main(void)
{
    int limit;
    int num = 20;
    int values[num];
    scanf("%d", &limit);

    for (int i = 0; i < limit; i++)
    {
        scanf("%d", &values[i]);
    }
    for (int i = limit - 1; i >= 0; i--)
    {
        printf("%d ", values[i]);
    }
    printf("\n");
    return 0;
}
