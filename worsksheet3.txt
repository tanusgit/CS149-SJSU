#include <stdio.h>
//15 20 0 5 -1
//10 20
int main(void)
{
    int sum = 0;
    int num;
    int avg;
    int max = 0;
    int count = 0;

    while (1)
    {   //taking user input
        scanf("%d", &num);
        if (num < 0)
            break;
        sum = sum + num;
        count++;
        if (num > max)
        {
            max = num;
        }
    }
    //calculating average
    avg = sum / count;
    //printing average and max
    printf("%d %d\n", avg, max);
    return 0;
}

#include <stdio.h>
int main(void)
{
    int limit;
    int num = 20;
    int values[num];
    scanf("%d", &limit);

    for (int i = 0; i < limit; i++)
    {   //taking user input
        scanf("%d", &values[i]);
    }
    for (int i = limit - 1; i >= 0; i--)
    {   //printing values in the reverse order
        printf("%d ", values[i]);
    }
    printf("\n");
    return 0;
}

#include <stdio.h>
//input = 5 50 60 140 200 75 100
//output = 50 60 75
int main()
{
    const int num = 20;
    int values[num];
    int i = 0;
    int start = 0;
    while (1)
    { //taking user input
        scanf("%d", &values[i]);
        //storing the number of values to chcek inside start
        start = values[0];
        if (i >= start + 1)
        {
            break;
        }
        i++;
    }
    // printf("%d %d\n", start, values[i]);
    for (int k = 1; k <= start; k++)
    {
        //checking if all the values except the first value is
        //less then the last value user entered
        if (values[k] < values[i])
        {
            //printing the values
            printf("%d ", values[k]);
        }
    }
    printf("\n");
    return 0;
}
