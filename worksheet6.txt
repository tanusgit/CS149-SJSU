#include <stdio.h> 
#include <string.h> 

void isPalindrome(char str[]) 
{  

int l = 0; 
int h = strlen(str) - 1; 

while (h > l) 
{
if(str[h]==' ')
{
h=h-1;
}
else if(str[l]==' ')
{
l=l+1;
}
else if (str[l++] != str[h--]) 
{ 
printf("%s is not a palindrome\n", str); 
return; 
} 
} 
printf("%s is a palindrome\n", str); 
} 

int main() 
{ 
    char input[1000];
    //getting user input
    gets(input);

isPalindrome(input); 
return 0; 

}

#include<stdio.h>
void SortArray(int arr[],int n)
{
int i;
int j;
for(i=0;i<n-1;i++)
{
for(j=0;j<n-i-1;j++)
{
if(arr[j]>arr[j+1])
{
int temp=arr[j];
arr[j]=arr[j+1];
arr[j+1]=temp;
}
}

}

}

{
int n;

int arr[n];
